[docs]def get_view_window(obj, default_view, default_name="NO_NAME"):
    view=getattr(obj, "view_window", None)
    if view is None:
        view=default_view(agent=obj)
    view.name=getattr(obj, "name", default_name)
    if view.title=="":
        view.title=view.name
    return view
